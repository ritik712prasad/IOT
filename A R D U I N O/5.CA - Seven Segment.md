# CC - Seven Segment
---
#### By using common anode(CA) displaing 0-9 with arduino Uno R3 
1. For that Set 7 segment as anode
2. All inputs are set by default to a high state, Resulting in LED lights being turned off.(see at starting time in led
3. For HIGH -> OFF LED, For LOW -> ON LED.

```c++
// C++ code
//displaing 0-9 in CA-seven segment display with arduino uno R3
//declared  all variables for inputs(with pin attached respectivelly)
//All inputs are set by default to a high state.
//Resulting in LED lights being turned off.
//For HIGH -> OFF LED, For LOW -> ON LED, 
int A=2;
int B=3;
int C=4;
int D=5;
int E=6; 
int F=7;
int G=8;

void setup()
{
   //set all seven pins for OUTPUT
   pinMode(A, OUTPUT);
   pinMode(B, OUTPUT);
   pinMode(C, OUTPUT);
   pinMode(D, OUTPUT);
   pinMode(E, OUTPUT);
   pinMode(F, OUTPUT);
   pinMode(G, OUTPUT);
}

void loop()
{
   
   // To  Print 0 (only set HIGH to pin G)
   delay(2000); // Wait for 2000 millisecond(s)
  
   digitalWrite(A, LOW);
   digitalWrite(B, LOW);
   digitalWrite(C, LOW);
   digitalWrite(D, LOW);
   digitalWrite(E, LOW);
   digitalWrite(F, LOW);
   digitalWrite(G, HIGH);
  
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 1 (only set LOW pin B and C)
   digitalWrite(A, HIGH);
   digitalWrite(B, LOW);
   digitalWrite(C, LOW);
   digitalWrite(D, HIGH);
   digitalWrite(E, HIGH);
   digitalWrite(F, HIGH);
   digitalWrite(G, HIGH);
  
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 2 (only set HIGH to pin C and F)
   digitalWrite(A, LOW);
   digitalWrite(B, LOW);
   digitalWrite(C, HIGH);
   digitalWrite(D, LOW);
   digitalWrite(E, LOW);
   digitalWrite(F, HIGH);
   digitalWrite(G, LOW);
  
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 3 (only set HIGH to pin E and F
   digitalWrite(A, LOW);
   digitalWrite(B, LOW);
   digitalWrite(C, LOW);
   digitalWrite(D, LOW);
   digitalWrite(E, HIGH);
   digitalWrite(F, HIGH);
   digitalWrite(G, LOW);
 
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 4 (only set HIGH to pin A, D and E)
   digitalWrite(A, HIGH);
   digitalWrite(B, LOW);
   digitalWrite(C, LOW);
   digitalWrite(D, HIGH);
   digitalWrite(E, HIGH);
   digitalWrite(F, LOW);
   digitalWrite(G, LOW);
  
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 5 (only set HIGH to pin B and E)
   digitalWrite(A, LOW);
   digitalWrite(B, HIGH);
   digitalWrite(C, LOW);
   digitalWrite(D, LOW);
   digitalWrite(E, HIGH);
   digitalWrite(F, LOW);
   digitalWrite(G, LOW);
  
   delay(1000); // Wait for 1000 millisecond(s)\
  
   // to  print 6 (only set HIGH to pin B)
   digitalWrite(A, LOW);
   digitalWrite(B, HIGH);
   digitalWrite(C, LOW);
   digitalWrite(D, LOW);
   digitalWrite(E, LOW);
   digitalWrite(F, LOW);
   digitalWrite(G, LOW);

   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 7 (only set LOW to pin A, B and C)
   digitalWrite(A, LOW);
   digitalWrite(B, LOW);
   digitalWrite(C, LOW);
   digitalWrite(D, HIGH);
   digitalWrite(E, HIGH);
   digitalWrite(F, HIGH);
   digitalWrite(G, HIGH);
 
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 8 (only set all HIGH)
   digitalWrite(A, LOW);
   digitalWrite(B, LOW);
   digitalWrite(C, LOW);
   digitalWrite(D, LOW);
   digitalWrite(E, LOW);
   digitalWrite(F, LOW);
   digitalWrite(G, LOW);
   
   delay(1000); // Wait for 1000 millisecond(s)
  
   // to  print 9 (only set HIGH to pin E)
  digitalWrite(A, LOW);
  digitalWrite(B, LOW);
  digitalWrite(C, LOW);
  digitalWrite(D, LOW);
  digitalWrite(E, HIGH);
  digitalWrite(F, LOW);
  digitalWrite(G, LOW);
}
```
